       ?level:int -> ?experience:int -> ?loc:int * int -> unit -> t

constructor s h l e () constructs a new player module located at row 1, col 1, with strength s, health h, experience e, at level l.

val location : t -> int * int

location p is the current location of player p.

val health : t -> int

health p is the current health of player p.

val max_health : t -> int

max_health p is the maximum health of a player p

val experience : t -> int

experience p is the current experience value of player p.

val level : t -> int

level p is the current level of player p

val col : t -> int

col p is the current col coordinate of player p.

val row : t -> int

row p is the current row coordinate of player p.

val move : t -> Maps.t -> int -> int -> unit

move p m col_diff row_diff moves player t to col_diff right and row_diff up on map m

val move_left : t -> Maps.t -> unit

move_left p m moves player p to (col-1, row) within the bounds of map m. The player p stays in its place when such a move causes him to move out of bounds.

val move_right : t -> Maps.t -> unit

move_right p m moves player p to (col+1, row) within the bounds of map m. The player p stays in its place when such a move causes him to move out of bounds.

val move_up : t -> Maps.t -> unit

move_up p m moves player p to (col, row+1) within the bounds of map m. The player p stays in its place when such a move causes him to move out of bounds.

val move_down : t -> Maps.t -> unit

move_down p m moves player p to (col, row-1) within the bounds of map m. The player p stays in its place when such a move causes him to move out of bounds.

val increase_health : t -> int -> unit

increase_health p h increases player p's health by h.

val increase_strength : t -> int -> unit

increase_strength p s increases player p's strength by s.

val reduce_health : t -> int -> unit

reduce_health p h reduces the health of player p by h. Requires: h >= 0

val reduce_strength : t -> int -> unit

reduce_strength p s reduces the strength of player p by s. Requires: s >= 0

val level_up_expereince : t -> int

level_up_expereince p is the experience required to level up player p

val advance_level : t -> unit

advance_level p advances player p to the next level and updates player p's experience as well. If p does not have enough experience to advance, no change will occur.

val increase_experience : t -> int -> unit

increase_experience p e increases the experience of player p by e. Requires: e >= 0

val get_skill_by_skill_name : t -> string -> skill

get_skill_by_skill_name p n returns skill s with the name n of player p. Raises UnknownSkill ("skill name " ^ [n] ^ "does not exist") if the player p does not have the skill named n.

val available_skills_list : t -> skill list

available_skills_list p is the skills that the player p posesses.

val skill_name : skill -> string

skill_name s is the name of the skill s.

val skill_strength : skill -> int

skill_strength s is the strength of the skill s.

val skill_description : skill -> string

skill_description s is the description of the skill s.

val assert_skill_name_NOT_in_list : t -> string -> bool

assert_skill_name_NOT_in_list p n returns true if there are no skills in the skill list of player p with name n, and false otherwise.

val update_skill : t -> skill list -> unit

update_skill p lst appends the new skill list lst to the existing skills repertoire of player p

val switch_loc : t -> int * int -> unit

switch_loc p loc changes the location of player p to loc.

val choose_skill : t -> skill -> unit

choose_skill p s will update the cd of skill s in player p

val get_all_skill_format : t -> (string * int) list

get_all_skill_format s is a list of (skill name, skill cd)
