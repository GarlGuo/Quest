val text_init : unit -> unit

text_init() loads the graphics data for map Require: graph must be opened

val map_size_cal : Maps.t -> (int * int) * int

map_size_cal t scales the size of the map model object t to the size of the GUI screen and returns the dimensions of the scaled map w.r.t the GUI.

val picture_getter : string -> 'a -> Graphics.image

picture_getter s returns the image RGB matrix given by the picture name s. Fails with "wrong name of picture, broken json or code" if s is not a valid picture name.

val map_text_build : unit -> unit

map_text_build () draws the map on the GUI based on the map params and returns ()

val draw_player : unit -> unit

draw_player () draws the player on the GUI and returns ().

val draw_weapon : int -> int -> int -> Builder.weapon_item -> unit

draw_weapon rr cr interval w draws everything in w with rr,cr as the lower-left point and each points have a interval of interval Requires: rr,cr,interval are positive int or 0 w is a vaild weapon list

val draw_food : int -> int -> int -> Builder.food_item -> unit

val draw_enemy : int -> int -> int -> Builder.enemy -> unit

val draw_items : unit -> unit

draw_items() draws every items, including foods, weapons, enemies, and portals, on the map
